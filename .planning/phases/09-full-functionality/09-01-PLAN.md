  bookings: { traveller_name: string; checkin_date: string; checkout_date: string } | null

**Fetch in useEffect:**
```ts
useEffect(() => {
  const supabase = createClient()
  supabase
    .from("feedback")
    .select(`
      id,
      computed_score,
      comment,
      created_at,
      bookings (
        traveller_name,
        checkin_date,
        checkout_date
      ),
      hotels (
        name
      )
    `)
    .order("created_at", { ascending: false })
    .then(({ data }) => {
      if (data && data.length > 0) {
        setFeedbackList(data as FeedbackRow[])
        setSelectedId(data[0].id)
      }
      setLoading(false)
    })
}, [])
```
